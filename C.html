题目1: 矩阵主对角线元素之和


//输入3x3矩阵，求该矩阵主对角线上的元素之和
#include<stdio.h>
int main() {
    int a[3][3] = { 0 }; //初始化3x3矩阵
    for (int i = 0; i < 3; i++) {
        for (int j = 0; j < 3; j++) {
            printf("请输入第%d行，第%d列的元素\n",i+1,j+1);
            scanf_s("%d",&a[i][j]);
        }
    }
    int sum = 0;
    for (int i = 0; i < 3; i++) {
        sum += a[i][i];
    }
    printf("3x3矩阵的主对角线元素和为:%d\n",sum);
}

三个数最大者


//用函数实现求三个数中的最大者
#include< stdio.h >
int max(int,int,int);
int main()
{
	int a = 0, b = 0, c = 0;
	printf("请输入三个数：\n");
	scanf_s("%d%d%d", &a, &b, &c);
	printf("三个数中最大的为：%d",max(a,b,c));
	return 0;
}
int max(int a, int b, int c)
{
	if (a > b)
		if (a > c)
			return a;
		else return c;
	else if(b > a)
		if (b > c)
			return b;
		else return c;
	return -1;
}

//写一个函数，判断一个数是否为“水仙花”，各位数字的立方和等于本身
#include
int main()
{
	int a = 0;
	printf("请输入数(三位数)：\n");
	scanf_s("%d",&a);
	int temp = 0;
	temp = (a % 10) * (a % 10) * (a % 10) + (a / 100) * (a / 100) * (a / 100) + (a % 100 / 10) * (a % 100 / 10) * (a % 100 / 10);
	if (temp == a&&temp!=0&&a!=0)
		printf("%d是水仙花数\n",a);
	else
		printf("%d不是水仙花数\n", a);
	return 0;
}
